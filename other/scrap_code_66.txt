GLfloat* create_uv_for_plane(const int width, const int height) {
	const GLfloat uv[plane_uv_floats] = {
		0.0f, 0.0f,
		width, 0.0f,
		0.0f, height,

		0.0f, height,
		width, height,
		width, 0.0f
	};

	GLfloat* const uv_data = malloc(plane_uv_bytes);
	memcpy(uv_data, uv, plane_uv_bytes);
	return uv_data;
}
